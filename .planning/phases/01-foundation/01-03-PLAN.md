    **Bundle size check script (`pwa/scripts/check-bundle-size.js`):**
    ```javascript
    #!/usr/bin/env node
    /**
     * CI gate: Verifies initial JS bundle < 200KB gzipped.
     * Run after `npm run build`.
     * Exit 0 = pass, Exit 1 = fail.
     */
    import { readdirSync, statSync } from 'fs';
    import { join } from 'path';
    import { execSync } from 'child_process';

    const BUILD_DIR = join(import.meta.dirname, '..', 'build');
    const MAX_SIZE_KB = 200;

    // Find all JS files in the build output
    function findJsFiles(dir, files = []) {
      for (const entry of readdirSync(dir, { withFileTypes: true })) {
        const fullPath = join(dir, entry.name);
        if (entry.isDirectory()) {
          findJsFiles(fullPath, files);
        } else if (entry.name.endsWith('.js')) {
          files.push(fullPath);
        }
      }
      return files;
    }

    try {
      const jsFiles = findJsFiles(BUILD_DIR);
      let totalSize = 0;

      for (const file of jsFiles) {
        const stat = statSync(file);
        totalSize += stat.size;
      }

      // Estimate gzipped size (typical compression ratio ~3:1 for JS)
      // For accuracy, check if .gz files exist from precompress
      let gzippedSize = 0;
      const gzFiles = findGzFiles(BUILD_DIR);

      if (gzFiles.length > 0) {
        for (const file of gzFiles) {
          if (file.endsWith('.js.gz')) {
            gzippedSize += statSync(file).size;
          }
        }
      } else {
        // Fallback: estimate
        gzippedSize = Math.round(totalSize / 3);
      }

      const gzippedKB = Math.round(gzippedSize / 1024);

      console.log(`JS bundle size: ${Math.round(totalSize / 1024)}KB raw, ~${gzippedKB}KB gzipped`);
      console.log(`Budget: ${MAX_SIZE_KB}KB gzipped`);

      if (gzippedKB > MAX_SIZE_KB) {
        console.error(`FAIL: Bundle exceeds ${MAX_SIZE_KB}KB gzipped budget (${gzippedKB}KB)`);
        process.exit(1);
      }

      console.log('PASS: Bundle within budget');
      process.exit(0);
    } catch (err) {
      console.error('Error checking bundle size:', err.message);
      process.exit(1);
    }

    function findGzFiles(dir, files = []) {
      for (const entry of readdirSync(dir, { withFileTypes: true })) {
        const fullPath = join(dir, entry.name);
        if (entry.isDirectory()) {
          findGzFiles(fullPath, files);
        } else if (entry.name.endsWith('.gz')) {
          files.push(fullPath);
        }
      }
      return files;
    }
    ```
